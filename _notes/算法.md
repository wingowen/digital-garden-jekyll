# 数组

**数组是存放在连续内存空间上的相同类型数据的集合。**

正是因为数组在内存空间的地址是连续的，所以我们在删除或者增添元素的时候，就难免要移动其他元素的地址。故数组的元素是不能删的，只能覆盖。

经典题目解法
- 二分法：循环变量不变原则。
- 双指针法：通过一个快指针和慢指针在一个 for 循环下完成两个 for 循环的工作。
- 滑动窗口：双指针的变式，重点在于根据当前子序列和大小的情况，不断调节子序列的起始位置。（==用于调节流？==）
- 前缀和：一次遍历求得区间和数组。

```c
// 有序的无重复元素数组

// 写法一：target 在 [left, right] 之间
Class Sloution{
public:
	int search()(vector<int>& nums, int target) {
		int left = 0;
		int right = nums().size - 1;
		while (left <= right) { // left == right 时 target 区间依然成立
			int middle = (left + right) / 2;
			if (nums[middle] > target){
				right = middle - 1;
			} else if (nums[middle] < target) {
				left = middle + 1;
			} else {
				return middle;
			}
		}
		return -1;
	}
};
```

```c
// 示例：给定 nums = [0,1,2,2,3,0,4,2], val = 2, 函数应该返回新的长度 5, 并且 nums 中的前五个元素为 0, 1, 3, 0, 4

// 暴力解法 时间复杂度为 O(n^2)
Class Solution{
public:
	int removeElement(vector<int>& nums, int val){
		int size = nums.size();
		for (int i = 0; i < size; i++){
			if(nums[i] == val){ // 发现需要移除的元素

				for (int j = i+1, j < size, j++){

				size--; // 数组减小一位
			}
		}
		return size;
	}
};
```

## 有序数组平方

给你一个按非递减顺序排序的整数数组 nums，返回每个数字的平方组成的新数组，要求也按非递减顺序排序。

```c
// 输入：nums = [-7,-3,2,3,11] 输出：[4,9,9,49,121]

// 暴力解法 直接平方后快排 O(n+nlogn)
class Solution{
public:
	vector<int> sortedSquares(vector<int>& nums){
		for (int i = 0; i < A.size(); i++){
			A[i] *= A[i];
		}
		sort(A.begin, A.end());
		return A;
	}
};

// 双指针法 O(n)
// 由于正负的存在 原数组有序 最大值一定在数组两端
// 利用左右指针进行平方后数组的排序
class Solution{
public:
	vector<int> sortedSquares(vector<int>& nums){
		int k = A.size() - 1;
		vector<int> result(A.size, 0);
		for(int i = 0; j = A.size() - 1 ; i <= j;){
			// 大的放尾部
			if(nums[i] * nums[i] > nums[j] * nums[j]){
				result[k--] = nums[i] * nums[i]
				i++;
			} else {
				result[k--] = nums[j] * nums[j]
				j--;
			}
		}
		return result;
	}
}
```

## 长度最小的子数组

给定一个含有 n 个正整数的数组和一个正整数 s ，找出该数组中满足其和 ≥ s 的长度最小的连续子数组，并返回其长度。如果不存在符合条件的子数组，返回 0。

```c
// 输入：s = 7, nums = [2,3,1,2,4,3]
// 输出：2 解释：子数组 [4,3] 是该条件下的长度最小的子数组。

// 暴力解法 两个 for 循环 一个控制起始位 一个控制终止位
class Solution{
public:
	int minSubArrayLen(int s, vector<int> nums){
		int result = INT32_MAX;
		int sum = 0; // 子数组总和
		int subLength = 0; // 子数组长度
		for (int i = 0; i < nums.size(); i++){
			sum = 0;
			for (int j = i; j < nums.size(); j++){
				sum += nums[j];
				if (sum >= s){ // 子数组满足条件
					subLength = j - i + 1;
					result = result > subLength? subLength : result;
					break;
				}
			}
		}
		return result == INT32_MAX? 0 : result;
	}

}

// 滑动窗口 其实也是双指针 一个起始指针 一个终止指针
class Solution{
public:
	int minSubArrayLen(int s, vector<int> nums){
		int result = INT32_MAX;
		int sum = 0; // 子数组总和
		int subLength = 0; // 子数组长度
		int i = 0;
		for (int j = 0; j < nums.size(); j++){
			sum += nums[j];
			while (sum >= s) {
				subLength = j - i + 1;
				result = subLength < result; subLength : result;
				sum -= nums[i--]; // 起始位向前滑动一位
			}
		}
	return result == INT32_MAX ; 0 : result;
	}
}

```

## 螺旋矩阵

![](assets/images/算法-1.png)

**循环要保证一致性**。每画一条边都要坚持一致的左闭右开，或者左开右闭的原则，这样这一圈才能按照统一的规则画下来。

```c
// 左开右闭 n 为奇数需要给中间单独赋值

class Solution {
public:
	vector<vector<int>> generateMatrix(int n){
		vector<vector<int>> res(n, vector<int>(n, 0));
		int startx = 0, starty = 0; // 定义每循环一个圈的起始位置
        int loop = n / 2; // 循环圈数
        int mid = n / 2; // 矩阵中间的位置
        int count = 1; // 用来给矩阵中每一个空格赋值
        int offset = 1; // 需要控制每一条边遍历的长度，每次循环右边界收缩一位
        int i,j;
        while (loop --) {
            i = startx;
            j = starty;

            // 下面开始的四个 for 就是模拟转了一圈
            // 模拟填充上行从左到右(左闭右开)
            for (j; j < n - offset; j++) {
                res[i][j] = count++;
            }
            // 模拟填充右列从上到下(左闭右开)
            for (i; i < n - offset; i++) {
                res[i][j] = count++;
            }
            // 模拟填充下行从右到左(左闭右开)
            for (; j > starty; j--) {
                res[i][j] = count++;
            }
            // 模拟填充左列从下到上(左闭右开)
            for (; i > startx; i--) {
                res[i][j] = count++;
            }

            // 下一圈起始位加一
            startx++;
            starty++;

            // offset 控制每一圈里每一条边遍历的长度
            offset += 1;
        }

        // 如果 n 为奇数的话，需要单独给矩阵最中间的位置赋值
        if (n % 2) {
            res[mid][mid] = count;
        }
        return res;		
	}
};
```

## 区间和

**前缀和**的思想是重复利用计算过的子数组之和，从而降低区间查询需要累加计算的次数。

```c
		int sum = 0;
		for (int i = a; i < b; i++) sum += vec[i];
```

## 开发商购买土地

由于城市规划的限制，只允许将区域按横向或纵向划分成两个子区域，而且每个子区域都必须包含一个或多个区块。 为了确保公平竞争，你需要找到一种分配方式，使得 A 公司和 B 公司各自的子区域内的土地总价值之差最小。

```c
// 如果将区域按照如下方式划分 前两列切割
// 1 2 | 3  
// 2 1 | 3  
// 1 2 | 3 
// 两个子区域内土地总价值之间的最小差距可以达到 0

// 使用前缀和的方式计算两个子区域的不平衡度 (sum - areaA) - areaA

#include <iostream>
#include <vector>
#include <climits>

int main(){
	int n, m;
	cin >> n >> m;
	int sum = 0;
	vector<vector<int>> vec(n, vector<int>(m, 0));
	// 全量统计
	for (int i = 0; i < n; i++){
		for (int j = 0; j < m; j++){
			cin >> vec[i][j];
			sum += vec[i][j];
		}
	}
	vector<int> hori(n, 0) // 横向统计
	for (int i = 0; i < n; i++){
		for (int j = 0; j < m; j++){
			hori[i] += vec[i][j];
		}
	}
	vector<int> vert(n, 0) // 纵向统计
	for (int j = 0; j < n; j++){
		for (int i = 0; i < m; i++){
			vert[j] += vec[i][j];
		}
	}
	int result = INT32_MAX;
	int horiCut = 0; // 横切统计
	for (int i = 0; i < n; i++){
		horiCut += heri[i];
		result = min(result, abs(sum - horiCut - horiCut)); // 切割第 n 行
	}
	int vertCut = 0; // 纵向统计
	for (int j = 0; j < n; j++){
		vertCut += vert[j];
		result = min(result, abs(sum - vertCut - vertCut)); // 切割第 n 列
	}
	cout << result << endl;
}
```

## 翻转链表

```c
// 双指针法
Class Solution{
public:
	ListNode* reverseList(ListNode* head){
		// 不断将 cur 指向 pre，即箭头指向左边
		ListNode* tmp;
		ListNode* cur = head;
		ListNode* pre = NULL;
		while(cur){
			tmp = cur -> next;
			cur -> next = pre;
			pre = cur;
			cur = tmp;
		}
		return pre;
	}
```
